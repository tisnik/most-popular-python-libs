def loop(x):
    while x > 0:
        x = x - 1
    return x

    

  

  

  

    

      

        #
      

      
Vse je nutne otestovat.

    

    

      
def main():
    print(loop(10))

def disassemble():
    from dis import dis

    print("\nloop:")
    dis(loop)


main()
