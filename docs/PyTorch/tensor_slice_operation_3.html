#

print(m1)

print()
#

konstrukce rezu - pres radky i sloupce

m2 = m1[1:3, 1:3]
print(m2)

print()
#

konstrukce rezu - pres radky i sloupce

m3 = m1[:3, :3]
print(m3)

print()
#

konstrukce rezu - pres radky i sloupce

m4 = m1[1:, 1:]
print(m4)

print()
