import json

import jq

with open("openapi.json") as fin:
    content = json.load(fin)

    for endpoint in jq.compile(".paths[]").input(content).all():
        print(",".join(endpoint.keys()))

    print("-------------------------")

    for has_get in jq.compile('.paths[] | has("get")').input(content).all():
        print(has_get)
