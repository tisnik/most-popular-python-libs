def generator1():
    values = ["a1", "b1", "c1", "d1"]
    for value in values:
        yield value

def generator3(suffix):
    values = ["a", "b", "c", "d", "e"]
    for value in values:
        yield "{v}{s}".format(v=value, s=suffix)


for v in _coconut.itertools.chain.from_iterable(
    (
        f()
        for f in (
            lambda: generator3("1"),
            lambda: generator3("2"),
            lambda: generator3("3"),
        )
    )
):
    print(v)
