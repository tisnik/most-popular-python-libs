from typing import Union, overload
#

Funkce s typovými anotacemi.

@overload
def add(a: int, b: int) -> int:
    ...


@overload
def add(a: str, b: str) -> str:
    ...


def add(a: Union[int, str], b: Union[int, str]) -> Union[int, str]:
# 
    return a + b
