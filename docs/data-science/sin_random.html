import numpy.random as rnd

r = rnd.random(100) - 0.5

x2 = np.linspace(np.pi, 3*np.pi, 100)

y2 = np.sin(x2)+r
#

vykreslit průběh obou funkcí

plt.plot(x1, y1, x2, y2)
#
